    public int lengthOfLongestSubstring(String s) {
        int maxLength=0;
        String longestSubString="";
        int a=0;
        for(int i=0;i<s.length();i++){
            longestSubString="";
            for(int j=i+1;j<s.length();j++){
                if(s.charAt(i)==s.charAt(j) || longestSubString.indexOf(s.charAt(j))>=0){
                    if((j-i)>maxLength){
                        maxLength = j-i;
                        //System.out.println(longestSubString+"..."+maxLength);
                    }
                    break;
                }else{
                    longestSubString = s.substring(i,j+1);
                }
            }
            //System.out.println(longestSubString+"..."+maxLength);
        }
        
        return maxLength;
    }

public class Solution {
    public int lengthOfLongestSubstring(String s) {
        boolean values[]=new boolean[256];
        int max_count=0,j=0,temp_count=0;
        char[] array=s.toCharArray();
        for(int i=j;i<array.length;i++){
            if(!values[array[i]]){
                temp_count++;
                values[array[i]]=true;
                if(temp_count>=max_count)
                    max_count=temp_count;
            }
            else{
                temp_count=0;
                values=new boolean[256];
                i=j;
                j++;
            }
        }
        return max_count;
    }
